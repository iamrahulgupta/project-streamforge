Protocol Specification

Overview

StreamForge uses Protocol Buffers (protobuf) for all binary communication. This provides:

  • Type safety and schema validation
  • Forward and backward compatibility
  • Efficient binary serialization
  • Language-independent protocol

Transport Layer

Connection Setup

Client                           Broker
  |                               |
  |---------- TCP SYN ---------->|
  |                               |
  |<------- TCP SYN-ACK ---------|
  |                               |
  |---------- TCP ACK ---------->|
  |                               |
  |------- METADATA_REQUEST ----->|
  |                               |
  |<----- METADATA_RESPONSE ------|
  |                               |

Message Frame Format

[Message Size: 4 bytes][Message Type: 1 byte][Correlation ID: 4 bytes][Payload: variable]

Example (hex): 00 00 01 00 | 01 | 00 00 00 00 | [protobuf data]
               ↑           ↑    ↑             ↑
         256 bytes     ProduceRequest  Correlation  Message content

Broker Protocol

ProduceRequest

message ProduceRequest {
  string topic = 1;
  repeated Message messages = 2;
  int32 required_acks = 3;    // -1=all, 0=none, 1=leader
  int32 timeout_ms = 4;
}

message Message {
  string key = 1;
  bytes value = 2;
  int64 timestamp = 3;
  int32 partition = 4;
  map headers = 6;
}

Example Request:

{
  "topic": "users",
  "messages": [
    {
      "key": "user-123",
      "value": "base64-encoded-data",
      "timestamp": 1646000000000,
      "partition": 0
    }
  ],
  "required_acks": 1,
  "timeout_ms": 30000
}

ProduceResponse

message ProduceResponse {
  int32 status = 1;           // 0=success, non-zero=error
  string error_message = 2;
  repeated PartitionResponse partition_responses = 3;
}

message PartitionResponse {
  int32 partition = 1;
  int64 base_offset = 2;      // First offset in batch
  int64 log_append_time = 3;  // Server timestamp
}

Example Response:

{
  "status": 0,
  "error_message": "",
  "partition_responses": [
    {
      "partition": 0,
      "base_offset": 1000000,
      "log_append_time": 1646000000000
    }
  ]
}

FetchRequest

message FetchRequest {
  string topic = 1;
  int32 partition = 2;
  int64 offset = 3;           // Starting offset
  int32 max_bytes = 4;        // Max bytes to return
  int32 max_wait_ms = 5;      // Max time to wait for data
}

FetchResponse

message FetchResponse {
  int32 status = 1;
  string error_message = 2;
  repeated Message messages = 3;
  int64 high_water_mark = 4;  // Latest offset available
}

Replication Protocol

AppendEntriesRequest

message AppendEntriesRequest {
  int32 leader_id = 1;
  int64 term = 2;             // Current term
  string topic = 3;
  int32 partition = 4;
  int64 prev_log_index = 5;   // Offset of log before new entries
  int64 prev_log_term = 6;
  repeated LogEntry entries = 7;
  int64 leader_commit = 8;    // Leader's commit index
}

message LogEntry {
  int64 offset = 1;
  int64 timestamp = 2;
  string key = 3;
  bytes value = 4;
  map headers = 5;
}

AppendEntriesResponse

message AppendEntriesResponse {
  int32 follower_id = 1;
  int64 term = 2;
  bool success = 3;
  int64 match_index = 4;      // Highest matched offset
  string error_message = 5;
}

HeartbeatRequest

message HeartbeatRequest {
  int32 leader_id = 1;
  int64 term = 2;
  string topic = 3;
  int32 partition = 4;
  int64 leader_commit = 5;
  int64 timestamp = 6;
}

Consumer Group Protocol

CommitOffsetRequest

message CommitOffsetRequest {
  string group_id = 1;
  string topic = 2;
  int32 partition = 3;
  int64 offset = 4;           // Offset to commit
}

CommitOffsetResponse

message CommitOffsetResponse {
  int32 status = 1;           // 0=success
  string error_message = 2;
}

FetchOffsetRequest

message FetchOffsetRequest {
  string group_id = 1;
  string topic = 2;
  int32 partition = 3;
}

FetchOffsetResponse

message FetchOffsetResponse {
  int32 status = 1;
  string error_message = 2;
  int64 offset = 3;           // Current committed offset
}

Metadata Protocol

MetadataRequest

message MetadataRequest {
  repeated string topics = 1; // Empty = all topics
}

message PartitionMetadata {
  int32 id = 1;
  int32 leader = 2;
  repeated int32 replicas = 3;
  repeated int32 isr = 4;     // In-Sync Replicas
}

MetadataResponse

message MetadataResponse {
  int32 status = 1;
  repeated BrokerMetadata brokers = 2;
  repeated TopicMetadata topics = 3;
}

message BrokerMetadata {
  int32 id = 1;
  string host = 2;
  int32 port = 3;
}

message TopicMetadata {
  string name = 1;
  int32 num_partitions = 2;
  int32 replication_factor = 3;
  repeated PartitionMetadata partitions = 4;
}

Message Format Details

Key Serialization

Keys support both string and binary formats:

  • String: UTF-8 encoded
  • Binary: Raw bytes
  • Null: Empty key for broadcast messages

Value Serialization

Values can be any binary data:

  • JSON documents
  • Protocol Buffer messages
  • Apache Avro
  • Raw binary
  • Compressed data

Timestamps

All timestamps are milliseconds since Unix epoch:

timestamp_ms = System.currentTimeMillis()  // Java
timestamp_ms = time.Now().UnixNano() / 1e6 // Go
timestamp_ms = int(time.time() * 1000)      // Python

Compression

Messages can be compressed to reduce network bandwidth:

Supported Codecs

Codec Code Ratio Speed
None 0 - Fastest
Gzip 1 5-10x Slow
Snappy 2 2-4x Fast
LZ4 3 3-5x Very Fast

Compression Headers

[Compression Codec: 1 byte][Compressed Data Size: 4 bytes][Compressed Data]

Handshake Sequence

Producer Handshake

1. Client connects to broker
2. Client sends MetadataRequest for topic
3. Broker responds with broker list and partition leaders
4. Client connects to partition leader
5. Client sends ProduceRequest
6. Broker responds with ProduceResponse

Consumer Handshake

1. Client connects to broker
2. Client sends MetadataRequest for topics
3. Client joins consumer group
4. Broker coordinates group rebalance
5. Broker assigns partitions to consumer
6. Client fetches metadata for assigned partitions
7. Client sends FetchRequest with current offset
8. Broker responds with FetchResponse

Error Handling

Retry Logic

def send_with_retry(request, max_retries=3):
    for attempt in range(max_retries):
        try:
            response = send(request, timeout=30s)
            if response.status == 0:
                return response
            elif is_retriable(response.status):
                wait(exponential_backoff(attempt))
                continue
            else:
                raise NonRetriableError(response)
        except TimeoutError:
            wait(exponential_backoff(attempt))
            continue
    raise MaxRetriesExceeded()

Idempotence

Producers can enable idempotent writes to guarantee exactly-once semantics:

producer_id = generate_unique_id()
sequence_number = 0

for message in messages:
    request.producer_id = producer_id
    request.sequence = sequence_number++
    send(request)

Examples

Produce with Custom Headers

message := &Message{
  Key:   []byte("order-123"),
  Value: []byte(jsonOrder),
  Headers: map[string]string{
    "source": "web-api",
    "version": "1.0",
    "correlation-id": "req-456",
  },
}

request := &ProduceRequest{
  Topic:        "orders",
  Messages:     []*Message{message},
  RequiredAcks: -1,   // Wait for all replicas
  TimeoutMs:    30000,
}

Fetch with Compression

request := &FetchRequest{
  Topic:        "logs",
  Partition:    0,
  Offset:       100000,
  MaxBytes:     1048576,      // 1MB
  MaxWaitMs:    5000,          // Wait up to 5s for data
}

// Broker will respond with compressed messages if beneficial

Performance Considerations

  • Message Batching: Reduces network roundtrips
  • Compression: Reduces bandwidth at cost of CPU
  • Connection Pooling: Reuse connections for lower latency
  • Async I/O: Non-blocking for higher throughput
  • Protocol Buffers: Efficient binary format
