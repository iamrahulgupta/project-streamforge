Topic Management

Create Topic

POST /api/v1/topics
Content-Type: application/json

{
  "name": "my-topic",
  "partitions": 3,
  "replication_factor": 2,
  "config": {}
}

{

    "name": "my-topic",
    "partitions": 3,
    "replication_factor": 2,

Go SDK API

Producer API

Create Producer

package main

import "github.com/iamrahulgupta/streamforge/sdk-go/producer"

config := producer.DefaultProducerConfig(
  []string{"localhost:9092"},
  "my-topic",
)
config.BatchSize = 100
config.BatchTimeoutMs = 100

p, err := producer.NewProducer(config)

Synchronous Produce

msg := &producer.Message{
  Key:   []byte("user-123"),
  Value: []byte("event data"),
}

result, err := p.ProduceSync(ctx, msg)
if err != nil {
  log.Fatal(err)
}

fmt.Printf("Offset: %d, Partition: %d\n", result.Offset, result.Partition)

Asynchronous Produce

resultChan, err := p.Produce(ctx, msg)

go func() {
  result := <-resultChan
  fmt.Printf("Message sent to offset %d\n", result.Offset)
}()

Consumer API

Create Consumer

config := consumer.DefaultConsumerConfig(
  []string{"localhost:9092"},
  "my-group",
  []string{"my-topic"},
)
config.AutoCommit = true
config.AutoCommitIntervalMs = 1000

c, err := consumer.NewConsumer(config)

Poll Messages

msg, err := c.Poll(ctx)
if err != nil {
  log.Fatal(err)
}

fmt.Printf("Key: %s, Value: %s\n", msg.Key, msg.Value)
c.CommitMessage(msg)

Poll Multiple Messages

messages, err := c.PollMessages(ctx, 100)
if err != nil {
  log.Fatal(err)
}

for _, msg := range messages {
  fmt.Printf("Received: %s\n", msg.Value)
}

// Commit all messages
for _, msg := range messages {
  c.CommitMessage(msg)
}

Batch Operations

Producer Batching

// Configure batching
config.BatchSize = 1000
config.BatchTimeoutMs = 100

// Messages are automatically batched
for i := 0; i < 5000; i++ {
  msg := &producer.Message{
    Key:   []byte(fmt.Sprintf("msg-%d", i)),
    Value: []byte("data"),
  }
  p.Produce(ctx, msg)
  
  // Every 1000 messages or 100ms, batch is sent
}

Error Handling

// Producer error handling
msg := &producer.Message{
  Key:   []byte("key"),
  Value: []byte("value"),
}

result, err := p.ProduceSync(ctx, msg)
if err != nil {
  switch err {
  case context.DeadlineExceeded:
    log.Println("Request timeout")
  case producer.ErrQueueFull:
    log.Println("Queue full, retry later")
  default:
    log.Printf("Error: %v", err)
  }
}

// Consumer error handling
msg, err := c.Poll(ctx)
if err != nil {
  switch err {
  case consumer.ErrNoMessages:
    log.Println("No messages available")
  case consumer.ErrConsumerClosed:
    log.Println("Consumer closed")
  default:
    log.Printf("Error: %v", err)
  }
}

Admin CLI API

Topic Commands

# List topics
streamforge topic list

# Create topic
streamforge topic create events \
  --partitions 3 \
  --replication-factor 2

# Describe topic
streamforge topic describe events

# Delete topic
streamforge topic delete events

# Alter topic config
streamforge topic alter events \
  --config "retention.ms=604800000"

Message Commands

# Produce message
streamforge produce \
  --topic events \
  --key "user-123" \
  --value "event data"

# Consume messages
streamforge consume \
  --topic events \
  --group my-group \
  --max-messages 10 \
  --format json

Server Command

# Start REST API server
streamforge server --port 8080 \
  --brokers broker-1:9092,broker-2:9092,broker-3:9092

Error Codes

Code Message Description
0 Success Operation completed successfully
1 UnknownError Unexpected error
2 TopicNotFound Topic does not exist
3 BrokerNotAvailable Broker is not available
4 MessageTooLarge Message size exceeds limit
5 OffsetOutOfRange Requested offset is invalid
6 NotLeaderForPartition Broker is not partition leader
7 RequestTimedOut Request timed out

Rate Limits

  • Producer: No hard limit, depends on broker capacity
  • Consumer: No hard limit, depends on broker capacity
  • Admin API: 1000 requests/second per client
  • Message Size: Default 1MB, configurable up to 10MB

Timeout Configuration

Parameter Default Description
request.timeout.ms 30000 (30s) Broker request timeout
session.timeout.ms 10000 (10s) Consumer session timeout
heartbeat.interval.ms 3000 (3s) Consumer heartbeat interval
metadata.max.age.ms 300000 (5min) Metadata refresh interval
